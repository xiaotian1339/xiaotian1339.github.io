计算平均分 

#include<stdio.h>
main()
{
	float score,a,n;
	a=0,n=0;//变量初始化
	do//直到循环
	{
		printf("请输入您的成绩(以负数结束)：\n");
		scanf("%f",&score);
		a+=score;
		n++;

	}while(score>0);
	a=(a-score)/--n;//n先自减在运算
	printf("平均分为%.3f\n",a);
}
