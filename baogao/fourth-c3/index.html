第四次上机报告

Tc.Shi

自动售货机为客户提供各种饮料，饮料的价格有3种: 2元、3元、5元，投入2元可选择"农夫山泉"、“怡宝矿泉水"和"康师傅矿泉水"之一，投入3元可以选择"可口可乐”、“雪碧"和"果粒橙"之一，投入5元可以选择"奶荼”、“咖啡"和"酸奶"之一。编写程序，模拟用户向自动售货机投入钱币，并选择一种饮料，使用scanf.模拟用户投入钱币，使用getchar,函数模拟用户选择的饮料。(输入A，B.C代表所选择的饮料) 

//自动售货机
#include<stdio.h>
main()
{
	int money;//定义整型变量
	char select;
	printf("您投入的金额为(￥):\n");
	scanf("%d",&money);
	switch (money)//识别钱数
	{
	case 2://两元
		printf("您投入了2元，您可以选择：\n");
		printf("A.农夫山泉\nB.怡宝矿泉水\nC.康师傅矿泉水\n");
		printf("请键入您的选择:\n");//提示输入
		fflush(stdin);
		select = getchar();
		printf("您的选择为：%c\n",select);
			switch (select)
			{
			case'a':
			case'A':
				printf("你得到了NongFu Spring！\n");
				break;
			case'b':
			case'B':
				printf("你得到了怡宝矿泉水!\n");
				break;
			case'c':
			case'C':
				printf("你得到了康师傅矿泉水！\n");
				break;
			default:
				printf("你的选择错误，作为惩罚，机器吞币，请重新投币！\n");
				break;
			}
			break;
	case 3:
		printf("您投入了3元，您可以选择：\n");
		printf("A.Coca Cola\nB.雪碧\nC.果粒橙\n");
		printf("请键入您的选择:\n");
		fflush(stdin);
		select = getchar();
		printf("您的选择为：%c\n",select);
			switch (select)
			{
			case'a':
			case'A':
				printf("你得到了Coca Cola！\n");
				break;
			case'b':
			case'B':
				printf("你得到了雪碧!\n");
				break;
			case'c':
			case'C':
				printf("你得到了果粒橙！\n");
				break;
			default:
				printf("你的选择错误，作为惩罚，机器吞币，请重新投币！\n");
				break;
		   	}
			break;
	case 5:
		printf("您投入了5元，您可以选择：\n");
		printf("A.奶茶\nB.咖啡\nC.酸奶\n");
		printf("请键入您的选择:\n");
		fflush(stdin);
		select = getchar();
		printf("您的选择为：%c\n",select);
			switch (select)
			{
			case'a':
			case'A':
				printf("你得到了奶茶！\n");
				break;
			case'b':
			case'B':
				printf("你得到了咖啡!\n");
				break;
			case'c':
			case'C':
				printf("你得到了酸奶！\n");
				break;
			default:
				printf("你的选择错误，作为惩罚，机器吞币，请重新投币！\n");
				break;
			}
			break;
	default:
		printf("本机器只识别2，3，5元，请重新投币！\n");
		break;
	}
	printf("欢迎下次光临本售货机，拜拜！\n");
}
